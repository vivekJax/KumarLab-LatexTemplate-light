% =============================================================================
% FONT SETUP - Smart font system with automatic fallbacks
% =============================================================================

% Check if DIN Next LT Pro font is available (Jackson Lab's preferred font)
\IfFontExistsTF{DIN Next LT Pro}{
    \setsansfont{DIN Next LT Pro}
    \message{DIN Next LT Pro font found and loaded}
}{
    \setsansfont{Helvetica}
    \message{DIN Next LT Pro not found, using Helvetica fallback}
}

% Check if Whitman font is available (for main text)
\IfFontExistsTF{Whitman-RomanLF}{
    \setmainfont{Whitman-RomanLF}
    \message{Whitman font found and loaded}
}{
    \setmainfont{Times New Roman}
    \message{Whitman not found, using Times New Roman fallback}
}

% =============================================================================
% FONT COMMANDS - Easy-to-use font definitions with automatic fallbacks
% =============================================================================

% Smart font command: uses sans-serif font (DIN or Helvetica)
\newcommand{\smartfont}[1]{%
    \sffamily #1%
}

% Title font: 20pt, bold, smart font (DIN or Helvetica)
\newcommand{\titlefont}{\fontsize{20}{24}\selectfont\bfseries\smartfont}

% Author font: 10pt, bold, smart font
\newcommand{\authorfont}{\fontsize{10}{12}\selectfont\bfseries\smartfont}

% Affiliation font: 10pt, regular, smart font
\newcommand{\affiliationfont}{\fontsize{10}{12}\selectfont\smartfont}

% Abstract heading font: 10pt, bold, smart font
\newcommand{\abstractheadingfont}{\fontsize{10}{12}\selectfont\bfseries\smartfont}

% Abstract text font: 9pt, regular, smart font
\newcommand{\abstractfont}{\fontsize{9}{11}\selectfont\smartfont}

% Metadata labels font: 8pt, bold, smart font (for "Code:", "Date:", etc.)
\newcommand{\metadatalabelfont}{\fontsize{8}{10}\selectfont\bfseries\smartfont}

% Metadata text font: 8pt, regular, smart font (for actual values)
\newcommand{\metadatatextfont}{\fontsize{8}{10}\selectfont\smartfont}

% Section title font: 16pt, bold, smart font
\newcommand{\sectionfont}{\fontsize{16}{18}\selectfont\bfseries\smartfont}

% Subsection title font: 14pt, bold, smart font
\newcommand{\subsectionfont}{\fontsize{14}{16}\selectfont\bfseries\smartfont}

% =============================================================================
% GREY BOX FOR TITLE PAGE
% =============================================================================

% Create a grey box with rounded corners for the title page
\newtcolorbox{titlebox}{
    colback=gray!10,        % Light grey background
    colframe=gray!10,       % Light grey border (same as background)
    boxrule=0pt,            % No visible border
    arc=8pt,                % Rounded corners (8 points radius)
    left=20pt,              % Left margin inside box
    right=20pt,             % Right margin inside box
    top=20pt,               % Top margin inside box
    bottom=20pt,            % Bottom margin inside box
    width=\textwidth        % Box width matches page width
}

% =============================================================================
% SECTION FORMATTING

% Format section titles with custom font

% Format subsection titles with custom font

% Format subsubsection titles with custom font
\titleformat{\subsubsection}{\small\bfseries\smartfont}{\thesubsubsection}{1em}{}

    citecolor=blue          % Citations are blue
}
